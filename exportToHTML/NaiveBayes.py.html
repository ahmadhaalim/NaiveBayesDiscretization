18               tempdf = self.__datalatih.loc[self.__datalatih.iloc[:, -1] == self.__uniqueClass[i]] 
19               tempdf = tempdf.reset_index(drop=True) 
20               tempprob = len(tempdf) / len(self.__datalatih) 
21               self.__classprob.update({self.__uniqueClass[i]: tempprob}) 
22               dictcolumn = {} 
23               #COLUMN LOOP 
24               j = 0 
25               while j<len(allcolumns)-1: 
26                   UniqueValues = self.__datalatih.__getattr__(allcolumns[j]).unique() 
27                   #UNIQUE COLUMN VALUE LOOP 
28                   tempdf2 = tempdf.iloc[:, [j, -1]] 
29                   dictvalue = {} 
30                   k = 0 
31                   while k< len(UniqueValues): 
32                       tempdf3 = tempdf2.loc[tempdf2[allcolumns[j]]==UniqueValues[k]] 
33                       tempdf3 = tempdf3.reset_index(drop=True) 
34                       valueprob = len(tempdf3)/len(tempdf) 
35                       dictvalue.update({UniqueValues[k]:valueprob}) 
36                       k+=1 
37    
38                   dictcolumn.update({j:dictvalue}) 
39                   j+=1 
40               self.__pxc.update({self.__uniqueClass[i]:dictcolumn}) 
41               #[CLASS][COLUMNS][VALUE] 
42               i += 1
